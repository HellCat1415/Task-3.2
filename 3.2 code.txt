#include "splashkit.h"

#define RADIUS 50 /* wouldnt reccoment defining radius in this way, since radiuses for circles can be different
i would reccomend rather doing a const variable, since radius is something that is subject to change, while PI and the accelaretion of gravity usually dont change, and those types of numbers are meant for defining. */

int main()
{
    double x, y;
    color circle_color = COLOR_GREEN;

    open_window("Circle Mover by Peta", 800, 600);

    x = screen_width() / 2;
    y = screen_height() / 2; /* i would reccomend using a different variable type for width and height of your program, unless you want half pixels lol */

    while( not quit_requested() )
    {
        process_events();
 
        if ( key_typed( C_KEY ) ) // i would reccomend using switches for this to increase readability
        {
            circle_color = random_color();
        }

        if ( key_down(RIGHT_KEY) )
        {
           x = x + 3; // look at next comment
        }

        if ( key_down(LEFT_KEY) )
        {
           x -= 3; // stick ot one style x-=y  or x=x-y take a look at my last comment and how your notation is not consistent.
        }

        if ( x + RADIUS > screen_width() )
        {
            x = screen_width() - RADIUS;
        }

        if ( x - RADIUS < 0 )
        {
            x = RADIUS;
        }

        if ( key_down(Q_KEY) )
        {
           y = y - 1;
        }

        if ( key_down(DOWN_KEY) )
        {
           y -= 1;
        }

        if ( y + RADIUS > screen_height() )
        {
            y = screen_height() - RADIUS;
        }

        if ( y - RADIUS < 0 )
        {
            y = RADIUS;
        }
        
        clear_screen(COLOR_WHITE);
        fill_circle(circle_color, x, 300, RADIUS);
        refresh_screen(60);

    }


    return 0;
}
